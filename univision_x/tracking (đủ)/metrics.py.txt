import numpy as np

def cosine_distance(a, b):
    return 1.0 - np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b) + 1e-6)

def iou(b1, b2):
    x1, y1, w1, h1 = b1
    x2, y2, w2, h2 = b2

    xa = max(x1 - w1/2, x2 - w2/2)
    ya = max(y1 - h1/2, y2 - h2/2)
    xb = min(x1 + w1/2, x2 + w2/2)
    yb = min(y1 + h1/2, y2 + h2/2)

    inter = max(0, xb - xa) * max(0, yb - ya)
    union = w1*h1 + w2*h2 - inter + 1e-6
    return inter / union
